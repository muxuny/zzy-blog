# 我的阅读与阅读历史 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为登录用户建立可追溯、可管理的阅读历史，并提供整合上次阅读、最近历史和收藏的“我的阅读”空间。

**Architecture:** 新增独立的阅读历史关系表、Mapper 和服务，公开文章详情只在成功读取后以非阻断方式触发历史写入；“我的阅读”聚合服务复用历史分页和现有收藏服务生成概览。前端新增鉴权路由、数据 API、纯函数工具、概览页和时间线历史页，公开首页、文章详情和收藏明细继续复用现有页面。

**Tech Stack:** Java 8、Spring Boot 2.7、Spring Security、MyBatis-Plus、MySQL 8、JUnit 5、Mockito、Vue 3、Vue Router、Element Plus、Axios、Node.js test runner、Vite

```java
private static String extractTableDefinition(String sql, String table,
                                             String comment, String source) {
    Pattern pattern = Pattern.compile(
            "CREATE\\s+TABLE\\s+IF\\s+NOT\\s+EXISTS\\s+`"
                    + Pattern.quote(table) + "`\\s*\\(.*?\\)\\s*ENGINE\\s*=\\s*InnoDB\\s+"
                    + "DEFAULT\\s+CHARSET\\s*=\\s*utf8mb4\\s+COLLATE\\s*=\\s*"
                    + "utf8mb4_unicode_ci\\s+COMMENT\\s*=\\s*'"
                    + Pattern.quote(comment) + "'\\s*;",
            Pattern.CASE_INSENSITIVE | Pattern.DOTALL);
    Matcher matcher = pattern.matcher(sql);
    assertTrue(matcher.find(), "Expected complete " + table + " table in " + source);
    return matcher.group();
}
```

```xml
<insert id="upsertHistory">
    INSERT INTO article_reading_history
        (id, user_id, article_id, title_snapshot, first_read_at, last_read_at, read_count,
         created_by, created_at, updated_by, updated_at, deleted, version)
    VALUES
        (#{id}, #{userId}, #{articleId}, #{titleSnapshot}, #{now}, #{now}, 1,
         #{username}, #{now}, #{username}, #{now}, 0, 0)
    ON DUPLICATE KEY UPDATE
        title_snapshot = VALUES(title_snapshot),
        first_read_at = IF(deleted = 1, VALUES(first_read_at), first_read_at),
        last_read_at = VALUES(last_read_at),
        read_count = IF(deleted = 1, 1, read_count + 1),
        updated_by = VALUES(updated_by),
        updated_at = VALUES(updated_at),
        version = version + 1,
        deleted = 0
</insert>

<update id="deleteHistory">
    UPDATE article_reading_history
    SET deleted = 1,
        updated_by = #{username},
        updated_at = #{now},
        version = version + 1
    WHERE user_id = #{userId}
      AND article_id = #{articleId}
      AND deleted = 0
</update>

<update id="clearHistory">
    UPDATE article_reading_history
    SET deleted = 1,
        updated_by = #{username},
        updated_at = #{now},
        version = version + 1

      AND deleted = 0
```

```xml
<select id="selectHistoryPage" resultType="com.blog.dto.ReadingHistoryRelationRow">
    SELECT h.id AS history_id,
           h.article_id,
           h.title_snapshot,
           h.first_read_at,
           h.last_read_at,
           h.read_count,
           CASE WHEN a.id IS NOT NULL
                     AND a.deleted = 0
                     AND a.status = 'published'
                     AND (a.visibility = 'public' OR a.visibility IS NULL)
                THEN 1 ELSE 0 END AS available
    FROM article_reading_history h
    LEFT JOIN article a ON a.id = h.article_id
    WHERE h.user_id = #{userId}
      AND h.deleted = 0
    ORDER BY h.last_read_at DESC, h.id DESC
</select>

<select id="selectLastAvailable" resultType="com.blog.dto.ReadingHistoryRelationRow">
    SELECT h.id AS history_id,
           h.article_id,
           h.title_snapshot,
           h.first_read_at,
           h.last_read_at,
           h.read_count,
           1 AS available
    FROM article_reading_history h
    INNER JOIN article a ON a.id = h.article_id
        AND a.deleted = 0
        AND a.status = 'published'
        AND (a.visibility = 'public' OR a.visibility IS NULL)
    WHERE h.user_id = #{userId}
      AND h.deleted = 0
    ORDER BY h.last_read_at DESC, h.id DESC
    LIMIT 1
</select>
```

```javascript
// utils/readingHistory.js
export function buildReadingHistoryParams({ page = 1, size = 10 } = {}) {
  const normalizedPage = Number(page)
  const normalizedSize = Number(size)
  return {
    page: Number.isInteger(normalizedPage) && normalizedPage > 0 ? normalizedPage : 1,
    size: Number.isInteger(normalizedSize) && normalizedSize > 0 && normalizedSize <= 100
      ? normalizedSize
      : 10
  }
}

export function formatReadingTime(value) {
  const date = new Date(value)
  if (Number.isNaN(date.getTime())) return ''
  return new Intl.DateTimeFormat('zh-CN', {
    year: 'numeric',
    month: 'short',
    day: 'numeric',
    hour: '2-digit',
    minute: '2-digit',
    hour12: false
  }).format(date)
}

function localDaySerial(value) {
  return Date.UTC(value.getFullYear(), value.getMonth(), value.getDate()) / 86400000
}

export function getReadingHistoryGroupKey(lastReadAt, now = new Date()) {
  const value = new Date(lastReadAt)
  if (Number.isNaN(value.getTime())) return 'earlier'
  const difference = localDaySerial(now) - localDaySerial(value)
  if (difference <= 0) return 'today'
  if (difference === 1) return 'yesterday'
  return 'earlier'
}

export function groupReadingHistory(items = [], now = new Date()) {
  const groups = [
    { key: 'today', label: '今天', items: [] },
    { key: 'yesterday', label: '昨天', items: [] },
    { key: 'earlier', label: '更早', items: [] }
  ]
  const byKey = new Map(groups.map(group => [group.key, group]))
  for (const item of items) {
    byKey.get(getReadingHistoryGroupKey(item.lastReadAt, now)).items.push(item)
  }
  return groups.filter(group => group.items.length)
}

export function getPageAfterHistoryDeletion({ page, size, total, deletedCount = 1 }) {
  const nextTotal = Math.max(0, Number(total) - Number(deletedCount))
  const maxPage = Math.max(1, Math.ceil(nextTotal / Number(size)))
  return Math.min(Math.max(1, Number(page)), maxPage)
}
```

- [ ] **Step 4: 创建历史页状态机**

`ReadingHistory.vue` 使用 `AppHeader`、`ReadingHistoryItem`、`getReadingHistory`、`deleteReadingHistory`、`clearReadingHistory`、`groupReadingHistory` 和 `getPageAfterHistoryDeletion`。脚本核心逻辑：

```javascript
const items = ref([])
const page = ref(1)
const size = ref(10)
const total = ref(0)
const loading = ref(false)
const loadError = ref('')
const clearing = ref(false)
const removingIds = ref(new Set())
let componentActive = true
let requestVersion = 0

const groups = computed(() => groupReadingHistory(items.value))

async function load() {
  const requestId = ++requestVersion
  loading.value = true
  loadError.value = ''
  try {
    const result = await getReadingHistory(buildReadingHistoryParams({
      page: page.value,
      size: size.value
    }))
    if (!componentActive || requestId !== requestVersion) return
    items.value = result.data || []
    total.value = Number(result.total) || 0
    const maxPage = Math.max(1, Math.ceil(total.value / size.value))
    if (page.value > maxPage) {
      page.value = maxPage
      await load()
    }
  } catch (error) {
    if (!componentActive || requestId !== requestVersion) return
    items.value = []
    total.value = 0
    loadError.value = error.response?.data?.message || error.message
      || '阅读历史加载失败，请重试'
  } finally {
    if (componentActive && requestId === requestVersion) loading.value = false
  }
}

async function removeHistory(item) {
  if (removingIds.value.has(item.articleId)) return
  removingIds.value = new Set(removingIds.value).add(item.articleId)
  try {
    await deleteReadingHistory(item.articleId)
    page.value = getPageAfterHistoryDeletion({
      page: page.value, size: size.value, total: total.value
    })
    ElMessage.success('已删除阅读历史')
    await load()
  } catch {
    // 共享请求拦截器已经显示失败信息，保留当前历史数据。
  } finally {
    const next = new Set(removingIds.value)
    next.delete(item.articleId)
    removingIds.value = next
  }
}

async function clearAllHistory() {
  try {
    await ElMessageBox.confirm(
      '清空后将移除全部阅读历史，重新阅读文章后会再次生成记录。',
      '清空阅读历史',
      { confirmButtonText: '清空', cancelButtonText: '取消', type: 'warning' }
    )
  } catch {
    return
  }
  clearing.value = true
  try {
    await clearReadingHistory()
    page.value = 1
    ElMessage.success('阅读历史已清空')
    await load()
  } catch {
    // 共享请求拦截器已经显示失败信息，保留当前历史数据。
  } finally {
    clearing.value = false
  }
}

function retryLoad() {
  void load()
}

function handlePageChange(nextPage) {
  page.value = nextPage
  void load()
}

function isRemoving(articleId) {
  return removingIds.value.has(articleId)
}
```

脚本顶部精确引入 `computed`、`onBeforeUnmount`、`onMounted`、`ref`、`ElMessage`、`ElMessageBox`、`ArrowLeft`、`Delete`、`Refresh`、`AppHeader`、`ReadingHistoryItem`、阅读 API 和阅读历史工具。`onMounted` 调用 `load`；`onBeforeUnmount` 设置 `componentActive=false` 并递增 `requestVersion`。
